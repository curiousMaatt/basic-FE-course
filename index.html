Some hints:

    git config --global user.name Name (реєстрація)
    git config --global user.email example@gmail.com (реєстрація)
    git init - ініціалізація репозиторію
    git status - інформація про файли
    git add [filename] - додати файл
    git commit - закоментувати файл (вперше git commit -m)
    git log - історія комітів
    git checkout - перехід між вітками
    git branch - робота з гілками
    git push - завантажити на сервер
    git pull - завантажити з сервера
    git merge - склеювання гілок
    git clone - копіювання репозиторію в нову директорію
